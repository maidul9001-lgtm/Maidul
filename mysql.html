CREATE TABLE products (
  id          INT AUTO_INCREMENT PRIMARY KEY,
  naam        VARCHAR(200) NOT NULL,
  price       DECIMAL(10,2) NOT NULL DEFAULT 0.00,
  stock       INT DEFAULT 0,
  category    ENUM('ইলেকট্রনিক্স','কাপড়','খাবার'),
  is_active   BOOLEAN DEFAULT TRUE,
  created_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP
              ON UPDATE CURRENT_TIMESTAMP
);

-- টেবিলের গঠন দেখা
DESCRIBE products;
SHOW CREATE TABLE products;
Foreign Key — টেবিল সম্পর্ক গুরুত্বপূর্ণ
▶

Foreign Key দিয়ে দুটো টেবিলের মধ্যে সম্পর্ক তৈরি হয়। যেমন orders টেবিলে customer_id থাকবে যা customers টেবিলের id-র সাথে মিলবে।

CREATE TABLE customers (
  id    INT AUTO_INCREMENT PRIMARY KEY,
  naam  VARCHAR(100) NOT NULL,
  phone VARCHAR(20) UNIQUE
);

CREATE TABLE orders (
  id           INT AUTO_INCREMENT PRIMARY KEY,
  customer_id  INT NOT NULL,
  total        DECIMAL(10,2),
  order_date   TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

  -- Foreign Key
  FOREIGN KEY (customer_id)
    REFERENCES customers(id)
    ON DELETE CASCADE
    ON UPDATE CASCADE
);

SELECT — তথ্য পড়া গুরুত্বপূর্ণ

-- সব তথ্য
SELECT * FROM products;

SELECT naam, price, stock FROM products;

-- কলামের নাম বদলে দেখানো
SELECT
  naam    AS 'পণ্যের নাম',
  price   AS 'মূল্য (টাকা)',
  stock   AS 'স্টক'
FROM products;

-- গণনা করে দেখানো
SELECT
  naam,
  price,
  price * 0.15 AS 'ভ্যাট',
  price * 1.15 AS 'মোট মূল্য'
FROM products;
WHERE দিয়ে ফিল্টার গুরুত্বপূর্ণ
▶ 
-- দাম ৫০০০ এর বেশি
SELECT * FROM products
WHERE price > 5000;

-- ইলেকট্রনিক্স ক্যাটাগরি
SELECT * FROM products
WHERE category = 'ইলেকট্রনিক্স';

-- AND দিয়ে
SELECT * FROM products
WHERE price > 1000 AND stock > 0;

SELECT * FROM products

SELECT * FROM products

-- LIKE দিয়ে খোঁজা
SELECT * FROM products
WHERE naam LIKE '%Samsung%';

-- NULL চেক
SELECT * FROM products
WHERE category IS NULL;

UPDATE — তথ্য পরিবর্তন গুরুত্বপূর্ণ
▶

⚠️ WHERE ছাড়া UPDATE করলে সব রেকর্ড পরিবর্তন হবে!

-- একটি পণ্যের দাম পরিবর্তন
UPDATE products
SET price = 43000.00
WHERE id = 1;

-- একসাথে অনেক কলাম
UPDATE products
SET price = 43000.00,
    stock = stock - 1,
    updated_at = NOW()
WHERE id = 1;

-- ১০% ছাড় দেওয়া সব পণ্যে
UPDATE products
SET price = price * 0.90
WHERE category = 'কাপড়';
DELETE — তথ্য মুছা গুরুত্বপূর্ণ
▶

⚠️ WHERE ছাড়া DELETE করলে সব তথ্য মুছে যাবে!

-- একটি পণ্য মুছা
DELETE FROM products
WHERE id = 5;

-- স্টক শূন্য যে পণ্যগুলো
DELETE FROM products
WHERE stock = 0;

-- Soft Delete (মুছে না, নিষ্ক্রিয় করা)
UPDATE products
SET is_active = FALSE
WHERE id = 5;

-- শুধু সক্রিয় পণ্য দেখা
SELECT * FROM products
WHERE is_active = TRUE;

Aggregate Functions গুরুত্বপূর্ণ
▶ 
-- মোট পণ্য সংখ্যা
SELECT COUNT(*) AS 'মোট পণ্য' FROM products;

-- সর্বোচ্চ ও সর্বনিম্ন দাম
SELECT
  MAX(price) AS 'সর্বোচ্চ',
  MIN(price) AS 'সর্বনিম্ন',
  AVG(price) AS 'গড়',
  SUM(price * stock) AS 'মোট মজুদ মূল্য'
FROM products
WHERE is_active = TRUE;
GROUP BY ও HAVING গুরুত্বপূর্ণ
▶ 
-- প্রতিটি ক্যাটাগরিতে কতটি পণ্য
SELECT
  category,
  COUNT(*) AS 'পণ্য সংখ্যা',
  AVG(price) AS 'গড় দাম',
  SUM(stock) AS 'মোট স্টক'
FROM products
GROUP BY category;

-- যে ক্যাটাগরিতে ৫টির বেশি পণ্য
SELECT category, COUNT(*) AS 'পণ্য'
FROM products
GROUP BY category
HAVING COUNT(*) > 5
ORDER BY COUNT(*) DESC;

INNER JOIN গুরুত্বপূর্ণ
▶

উভয় টেবিলে যা মেলে শুধু সেটাই দেখায়।

-- অর্ডার ও গ্রাহকের তথ্য একসাথে
SELECT
  o.id          AS 'অর্ডার নং',
  c.naam        AS 'গ্রাহক',
  c.phone       AS 'ফোন',
  o.total       AS 'মোট',
  o.order_date  AS 'তারিখ'
FROM orders o
INNER JOIN customers c
  ON o.customer_id = c.id
ORDER BY o.order_date DESC;
LEFT JOIN ও RIGHT JOIN
▶ 
-- সব গ্রাহক দেখাবে, অর্ডার না থাকলে NULL
SELECT
  c.naam AS 'গ্রাহক',
  COUNT(o.id) AS 'অর্ডার সংখ্যা',
  COALESCE(SUM(o.total), 0) AS 'মোট কেনাকাটা'
FROM customers c
LEFT JOIN orders o ON c.id = o.customer_id
GROUP BY c.id, c.naam
ORDER BY SUM(o.total) DESC;

-- যে গ্রাহকরা কখনো অর্ডার দেননি
SELECT c.naam
FROM customers c
LEFT JOIN orders o ON c.id = o.customer_id
WHERE o.id IS NULL;
তিনটি টেবিল একসাথে JOIN
▶ 
-- orders, customers, products একসাথে
SELECT
  o.id          AS 'অর্ডার',
  c.naam        AS 'গ্রাহক',
  p.naam        AS 'পণ্য',
  oi.quantity   AS 'পরিমাণ',
  oi.price      AS 'মূল্য'
FROM orders o
INNER JOIN customers c   ON o.customer_id = c.id
INNER JOIN order_items oi ON o.id = oi.order_id
INNER JOIN products p    ON oi.product_id = p.id;

Subquery গুরুত্বপূর্ণ
▶ 
-- গড় দামের বেশি যে পণ্যগুলো
SELECT naam, price
FROM products
WHERE price > (
  SELECT AVG(price) FROM products
);

-- সবচেয়ে বেশি অর্ডার করা গ্রাহক
SELECT naam
FROM customers
WHERE id = (
  SELECT customer_id
  FROM orders
  GROUP BY customer_id
  ORDER BY COUNT(*) DESC
  LIMIT 1
);
VIEW তৈরি ও ব্যবহার
▶ 
-- জনপ্রিয় পণ্যের View
CREATE VIEW popular_products AS
SELECT
  p.naam,
  p.price,
  COUNT(oi.id) AS 'বিক্রি'
FROM products p
INNER JOIN order_items oi ON p.id = oi.product_id
GROUP BY p.id
ORDER BY COUNT(oi.id) DESC;

-- View ব্যবহার
SELECT * FROM popular_products LIMIT 10;

-- View মুছা
DROP VIEW popular_products;

Stored Procedure অ্যাডভান্সড
▶

Stored Procedure হলো ডেটাবেসে সংরক্ষিত SQL কোড — ফাংশনের মতো বারবার কল করা যায়।

-- Procedure তৈরি
DELIMITER //
CREATE PROCEDURE GetProductsByCategory(
  IN cat_name VARCHAR(100)
)
BEGIN
  SELECT naam, price, stock
  FROM products
  WHERE category = cat_name
    AND is_active = TRUE
  ORDER BY price;
END //
DELIMITER ;

-- Procedure কল করা
CALL GetProductsByCategory('ইলেকট্রনিক্স');

-- Procedure মুছা
DROP PROCEDURE GetProductsByCategory;

TRIGGER — স্বয়ংক্রিয় কাজ অ্যাডভান্সড
▶

Trigger — কোনো ঘটনায় স্বয়ংক্রিয়ভাবে চলে। যেমন অর্ডার হলে স্টক কমবে।

DELIMITER //
CREATE TRIGGER after_order_insert
AFTER INSERT ON order_items
FOR EACH ROW
BEGIN
  -- অর্ডার হলে স্টক স্বয়ংক্রিয় কমবে
  UPDATE products
  SET stock = stock - NEW.quantity
  WHERE id = NEW.product_id;
END //
DELIMITER ;
