Database ও SQL কী?
▶

Database হলো তথ্য সংরক্ষণের গুদামঘর। SQL (Structured Query Language) দিয়ে সেই তথ্য যোগ, দেখা, পরিবর্তন ও মুছা যায়।

যেমন — একটি স্কুলের ডেটাবেসে ছাত্রছাত্রীর নাম, রোল, মার্কস সব টেবিলে সাজানো থাকে। SQL দিয়ে প্রশ্ন করে নির্দিষ্ট তথ্য বের করা যায়।

MySQL / SQLite ইনস্টল করা
▶

MySQL — সবচেয়ে জনপ্রিয়। SQLite — হালকা, ফাইলভিত্তিক (মোবাইলেও চলে)।

শুরু করতে DB Browser for SQLite (বিনামূল্যে) ডাউনলোড করুন — সহজ GUI আছে। অথবা অনলাইনে sqliteonline.com ব্যবহার করুন।

Table, Row, Column কী?
▶

ডেটাবেস Excel-এর মতো। Table = শিট, Column = কলাম (নাম, বয়স), Row = একটি রেকর্ড।

-- students টেবিলের উদাহরণ:
-- id | naam    | boyosh | marks
-- ---|---------|--------|------
--  1 | Maidul  |  25    |  85
--  2 | Rahim   |  22    |  90
--  3 | Karim   |  24    |  78

Database তৈরি ও ব্যবহার
▶ 
-- নতুন ডেটাবেস তৈরি
CREATE DATABASE school_db;

-- ডেটাবেস ব্যবহার শুরু
USE school_db;

-- সব ডেটাবেস দেখা
SHOW DATABASES;

-- ডেটাবেস মুছা
DROP DATABASE school_db;
Data Types — তথ্যের ধরন গুরুত্বপূর্ণ
▶ 
-- সংখ্যা
INT          -- পূর্ণ সংখ্যা: 1, 25, 1000
FLOAT        -- দশমিক: 3.14, 9.99
DECIMAL(8,2) -- নির্দিষ্ট দশমিক: 12345.67

-- টেক্সট
VARCHAR(100) -- সর্বোচ্চ ১০০ অক্ষর
TEXT         -- বড় লেখা
CHAR(10)     -- নির্দিষ্ট ১০ অক্ষর

-- তারিখ
DATE         -- 2024-01-15
DATETIME     -- 2024-01-15 10:30:00

-- সত্য/মিথ্যা
BOOLEAN      -- TRUE বা FALSE
Table তৈরি: CREATE TABLE গুরুত্বপূর্ণ
▶ 
CREATE TABLE students (
    id      INT PRIMARY KEY AUTO_INCREMENT,
    naam    VARCHAR(100) NOT NULL,
    boyosh  INT,
    email   VARCHAR(200) UNIQUE,
    marks   DECIMAL(5,2),
    join_date DATE DEFAULT CURRENT_DATE
);

-- টেবিলের গঠন দেখা
DESCRIBE students;

-- টেবিল মুছা
DROP TABLE students;

৩-৫ দিন ▼
INSERT — তথ্য যোগ করা গুরুত্বপূর্ণ
▶ 
-- একটি রেকর্ড যোগ
INSERT INTO students (naam, boyosh, email, marks)
VALUES ('Maidul', 25, 'maidul@gmail.com', 85.50);

-- একসাথে অনেক রেকর্ড
INSERT INTO students (naam, boyosh, marks)
VALUES
    ('Rahim',  22, 90.00),
    ('Karim',  24, 78.50),
    ('Sumaiya', 21, 95.00);
SELECT — তথ্য দেখা গুরুত্বপূর্ণ
▶ 
-- সব তথ্য দেখা
SELECT * FROM students;

-- নির্দিষ্ট কলাম
SELECT naam, marks FROM students;

-- কলামের নাম বদলে দেখানো
SELECT naam AS 'নাম', marks AS 'নম্বর'
FROM students;

-- ডুপ্লিকেট বাদ দিয়ে
SELECT DISTINCT boyosh FROM students;
WHERE — শর্ত দিয়ে ফিল্টার গুরুত্বপূর্ণ
▶ 
-- নির্দিষ্ট শর্তে দেখা
SELECT * FROM students
WHERE marks > 80;

-- AND দিয়ে একাধিক শর্ত
SELECT * FROM students
WHERE marks > 80 AND boyosh < 25;

-- OR দিয়ে
SELECT * FROM students
WHERE naam = 'Maidul' OR naam = 'Rahim';

-- IN দিয়ে তালিকা
SELECT * FROM students
WHERE naam IN ('Maidul', 'Rahim', 'Karim');

-- BETWEEN দিয়ে রেঞ্জ
SELECT * FROM students
WHERE marks BETWEEN 70 AND 90;
LIKE — আংশিক মিল খোঁজা
▶

% — যেকোনো কিছু | _ — যেকোনো একটি অক্ষর

-- 'M' দিয়ে শুরু
SELECT * FROM students
WHERE naam LIKE 'M%';

-- 'dul' দিয়ে শেষ
SELECT * FROM students
WHERE naam LIKE '%dul';

-- মাঝে 'ah' আছে
SELECT * FROM students
WHERE naam LIKE '%ah%';

-- gmail ব্যবহারকারী
SELECT * FROM students
WHERE email LIKE '%@gmail.com';
ORDER BY ও LIMIT
▶ 
-- নম্বর অনুযায়ী ছোট থেকে বড়
SELECT * FROM students
ORDER BY marks ASC;

-- নম্বর অনুযায়ী বড় থেকে ছোট
SELECT * FROM students
ORDER BY marks DESC;

-- শুধু প্রথম ৫ জন
SELECT * FROM students
ORDER BY marks DESC
LIMIT 5;

-- ৬ থেকে ১০ নম্বর রেকর্ড
SELECT * FROM students
LIMIT 5 OFFSET 5;

তথ্য পরিবর্তন ও মুছাUPDATE, DELETE
২-৩ দিন ▼
UPDATE — তথ্য পরিবর্তন করা গুরুত্বপূর্ণ
▶

⚠️ সতর্কতা: WHERE ছাড়া UPDATE লিখলে সব রেকর্ড পরিবর্তন হয়ে যাবে!

-- একটি রেকর্ড আপডেট
UPDATE students
SET marks = 90.00
WHERE id = 1;

-- একসাথে অনেক কলাম আপডেট
UPDATE students
SET marks = 95.00,
    boyosh = 26
WHERE naam = 'Maidul';

-- শর্ত অনুযায়ী অনেক রেকর্ড
UPDATE students
SET marks = marks + 5
WHERE marks < 60;
DELETE — তথ্য মুছা গুরুত্বপূর্ণ
▶

⚠️ সতর্কতা: WHERE ছাড়া DELETE লিখলে টেবিলের সব তথ্য মুছে যাবে!

-- একটি রেকর্ড মুছা
DELETE FROM students
WHERE id = 3;

-- শর্ত অনুযায়ী মুছা
DELETE FROM students
WHERE marks < 50;

-- টেবিলের সব তথ্য মুছা (সাবধান!)
DELETE FROM students;

-- দ্রুত মুছার উপায় (reset করে)
TRUNCATE TABLE students;

ALTER TABLE students

-- টেবিলের নাম পরিবর্তন
RENAME TABLE students TO shikkharthi;

Aggregate FunctionsCOUNT, SUM, AVG, MAX, MIN
৩-৪ দিন ▼
COUNT, SUM, AVG, MAX, MIN গুরুত্বপূর্ণ
▶ 
-- মোট ছাত্র সংখ্যা
SELECT COUNT(*) AS 'মোট ছাত্র' FROM students;

-- সর্বোচ্চ নম্বর
SELECT MAX(marks) AS 'সর্বোচ্চ' FROM students;

-- সর্বনিম্ন নম্বর
SELECT MIN(marks) AS 'সর্বনিম্ন' FROM students;

-- গড় নম্বর
SELECT AVG(marks) AS 'গড়' FROM students;

-- সব নম্বরের যোগফল
SELECT SUM(marks) AS 'যোগফল' FROM students;

-- একসাথে সব
SELECT
    COUNT(*) AS 'মোট',
    AVG(marks) AS 'গড়',
    MAX(marks) AS 'সর্বোচ্চ',
    MIN(marks) AS 'সর্বনিম্ন'
FROM students;
GROUP BY — গ্রুপ করে হিসাব গুরুত্বপূর্ণ
▶ 
-- ধরুন students টেবিলে class কলাম আছে
-- প্রতিটি ক্লাসে কতজন ছাত্র?
SELECT class, COUNT(*) AS 'ছাত্র সংখ্যা'
FROM students
GROUP BY class;

-- প্রতিটি ক্লাসের গড় নম্বর
SELECT class,
       AVG(marks) AS 'গড় নম্বর',
       COUNT(*) AS 'ছাত্র'
FROM students
GROUP BY class
ORDER BY AVG(marks) DESC;
HAVING — গ্রুপের উপর শর্ত
▶

WHERE → রেকর্ডের উপর শর্ত | HAVING → GROUP BY-এর পর শর্ত

-- যে ক্লাসে ১০ জনের বেশি ছাত্র
SELECT class, COUNT(*) AS 'সংখ্যা'
FROM students
GROUP BY class
HAVING COUNT(*) > 10;

-- গড় নম্বর ৭৫ এর বেশি যে ক্লাসে
SELECT class, AVG(marks) AS 'গড়'
FROM students
GROUP BY class
HAVING AVG(marks) > 75;

JOIN — টেবিল একত্রিত করাINNER, LEFT, RIGHT JOIN
১ সপ্তাহ ▼
JOIN কী এবং কেন দরকার?
▶

দুটি টেবিলের মিলিত তথ্য একসাথে দেখার জন্য JOIN ব্যবহার হয়।

-- students টেবিল: id, naam, class_id
-- classes  টেবিল: id, class_naam

-- এই দুটো মেলাতে JOIN লাগবে
INNER JOIN — উভয়ে মিলে যা আছে গুরুত্বপূর্ণ
▶ 
SELECT
    s.naam AS 'ছাত্রের নাম',
    c.class_naam AS 'ক্লাস',
    s.marks AS 'নম্বর'
FROM students s
INNER JOIN classes c
    ON s.class_id = c.id;

-- শুধু ৮০+ নম্বরপ্রাপ্তদের
SELECT s.naam, c.class_naam, s.marks
FROM students s
INNER JOIN classes c ON s.class_id = c.id
WHERE s.marks > 80
ORDER BY s.marks DESC;
LEFT JOIN ও RIGHT JOIN
▶

LEFT JOIN — বাম টেবিলের সব রেকর্ড, ডানে না মিললে NULL
RIGHT JOIN — ডান টেবিলের সব রেকর্ড, বামে না মিললে NULL

-- সব ছাত্র, ক্লাস না থাকলেও দেখাবে
SELECT s.naam, c.class_naam
FROM students s
LEFT JOIN classes c
    ON s.class_id = c.id;

-- যাদের ক্লাস নেই (NULL)
SELECT s.naam
FROM students s
LEFT JOIN classes c ON s.class_id = c.id
WHERE c.id IS NULL;

Subquery ও Viewক্যোয়েরির ভেতরে ক্যোয়েরি
৩-৫ দিন ▼
Subquery — ক্যোয়েরির ভেতরে ক্যোয়েরি গুরুত্বপূর্ণ
▶ 
-- গড়ের চেয়ে বেশি নম্বর পেয়েছে যারা
SELECT naam, marks
FROM students
WHERE marks > (
    SELECT AVG(marks) FROM students
);

-- সর্বোচ্চ নম্বরপ্রাপ্ত ছাত্র
SELECT naam, marks
FROM students
WHERE marks = (
    SELECT MAX(marks) FROM students
);
VIEW — ভার্চুয়াল টেবিল অ্যাডভান্সড
▶

VIEW হলো একটি সংরক্ষিত SELECT ক্যোয়েরি। বারবার জটিল ক্যোয়েরি না লিখে View ব্যবহার করা যায়।

-- View তৈরি
CREATE VIEW top_students AS
SELECT naam, marks
FROM students
WHERE marks >= 80
ORDER BY marks DESC;

-- View ব্যবহার (টেবিলের মতো)
SELECT * FROM top_students;

-- View মুছা
DROP VIEW top_students;

অ্যাডভান্সড ও পরবর্তী ধাপIndex, Transaction, Stored Procedure
১ সপ্তাহ ▼
INDEX — দ্রুত খোঁজার জন্য অ্যাডভান্সড
▶

বইয়ের সূচিপত্রের মতো — Index থাকলে ডেটা অনেক দ্রুত খোঁজা যায়।

-- Index তৈরি
CREATE INDEX idx_naam
ON students (naam);

-- Unique Index (ডুপ্লিকেট নেই)
CREATE UNIQUE INDEX idx_email
ON students (email);

-- Index দেখা
SHOW INDEX FROM students;

-- Index মুছা
DROP INDEX idx_naam ON students;
TRANSACTION — নিরাপদ অপারেশন অ্যাডভান্সড
▶

Transaction দিয়ে একাধিক কাজ একসাথে করা হয়। কোনো একটি ভুল হলে সব পূর্বাবস্থায় ফিরে যায়।

-- ব্যাংক ট্রান্সফারের উদাহরণ
START TRANSACTION;

UPDATE accounts
SET balance = balance - 500
WHERE id = 1;  -- A থেকে কাটা

UPDATE accounts
SET balance = balance + 500
WHERE id = 2;  -- B-তে যোগ

-- সফল হলে সেভ করো
COMMIT;

-- ভুল হলে ফিরিয়ে আনো
ROLLBACK;
উপযোগী SQL ফাংশন
▶ 
-- স্ট্রিং ফাংশন
SELECT UPPER(naam) FROM students;   -- বড় হাতে
SELECT LOWER(naam) FROM students;   -- ছোট হাতে
SELECT LENGTH(naam) FROM students;  -- অক্ষর সংখ্যা
SELECT CONCAT(naam, ' - ', marks) FROM students;

-- সংখ্যা ফাংশন
SELECT ROUND(marks, 1) FROM students; -- গোলাকার
SELECT ABS(-10);                        -- পরম মান → 10

-- তারিখ ফাংশন
SELECT NOW();                 -- বর্তমান সময়
SELECT YEAR(join_date) FROM students;
SELECT MONTH(join_date) FROM students;
