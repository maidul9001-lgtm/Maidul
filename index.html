Recursive Function (নিজেকে কল করে)
Recursive ফাংশন নিজেকে নিজে কল করে, একটি base case দিয়ে থামে।
# ফ্যাক্টোরিয়াল: n! = n × (n-1) × ... × 1
def factorial(n):
    if n == 0 or n == 1:  # base case
        return 1
    return n * factorial(n - 1)  # recursive call

print(factorial(5))  # 120 (5×4×3×2×1)

# ফিবোনাচি সিরিজ
def fib(n):
    if n <= 1:
        return n
    return fib(n-1) + fib(n-2)

print(fib(7))  # 13

class Dog:

Inheritance (উত্তরাধিকার) গুরুত্বপূর্ণ
একটি Class অন্য Class-এর সব বৈশিষ্ট্য গ্রহণ করতে পারে।
class Animal:          # Parent class
    def __init__(self, name):
        self.name = name

    def speak(self):
        print("কিছু শব্দ করছে...")

class Dog(Animal):     # Child class — Animal থেকে inherit
    def speak(self):   # Override
        print(f"{self.name} বলছে: ঘেউ ঘেউ!")

class Cat(Animal):
    def speak(self):
        print(f"{self.name} বলছে: মিয়াও!")

d = Dog("ভুলু")

class BankAccount:

        self.__balance = balance  # __ দিয়ে private

    def deposit(self, amount):
        if amount > 0:
            self.__balance += amount

    def get_balance(self):         # getter
        return self.__balance

class Shape:
    def area(self):
        pass

class Circle(Shape):
    def __init__(self, r):
        self.r = r
    def area(self):
        return 3.14 * self.r ** 2

class Rectangle(Shape):
    def __init__(self, w, h):
        self.w, self.h = w, h
    def area(self):
        return self.w * self.h

shapes = [Circle(5), Rectangle(4, 6)]

Magic/Dunder Methods (__str__, __len__)
Dunder (double underscore) methods Python-এর built-in আচরণ কাস্টমাইজ করে।
class Book:
    def __init__(self, title, pages):
        self.title = title
        self.pages = pages

    def __str__(self):      # print() করলে দেখায়
        return f"বই: {self.title}"

    def __len__(self):      # len() করলে দেখায়
        return self.pages

    def __repr__(self):     # debugging-এ দেখায়
        return f"Book('{self.title}', {self.pages})"

b = Book("Python শিখি", 350)
print(b)        # বই: Python শিখি
print(len(b))   # 350

Decorator ফাংশন অ্যাডভান্সড
Decorator হলো একটি ফাংশন যা অন্য ফাংশনকে মোড়ক দিয়ে তার আচরণ পরিবর্তন করে।
def timer(func):         # decorator
    def wrapper(*args, **kwargs):
        import time
        start = time.time()
        result = func(*args, **kwargs)
        end = time.time()
        print(f"{func.__name__} নিল {end-start:.4f} সেকেন্ড")
        return result
    return wrapper

@timer                  # এটি = my_func = timer(my_func)
def my_func():
    total = sum(range(1000000))
    return total

my_func()  # সময় দেখাবে
Generator ও yield কীওয়ার্ড অ্যাডভান্সড
Generator একবারে একটি মান তৈরি করে — বড় ডেটায় memory সাশ্রয়ী।
# সাধারণ ফাংশন — সব একসাথে মেমোরিতে রাখে
def squares_list(n):
    return [i**2 for i in range(n)]

# Generator — একটি একটি করে দেয়
def squares_gen(n):
    for i in range(n):
        yield i**2   # এখানে থামে, পরে আবার চলে

gen = squares_gen(5)
print(next(gen))  # 0
print(next(gen))  # 1

for val in squares_gen(5):
    print(val)  # 0, 1, 4, 9, 16

class MyContext:
    def __enter__(self):
        print("শুরু হচ্ছে...")

        print("শেষ হয়েছে!")
        return False  # exception suppress করবে না

with MyContext() as ctx:
