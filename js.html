JavaScript কী এবং কেন শিখবো?
▶

JavaScript একটি উচ্চ-স্তরের, ডায়নামিক প্রোগ্রামিং ভাষা। এটি শিখবেন কারণ:

• ওয়েব ব্রাউজারের একমাত্র প্রোগ্রামিং ভাষা
• ফ্রন্টএন্ড ও ব্যাকএন্ড (Node.js) — দুটোই করা যায়
• React, Vue, Angular — সব জনপ্রিয় ফ্রেমওয়ার্ক JS-এ
• চাকরির বাজারে সর্বোচ্চ চাহিদার ভাষাগুলোর একটি

ব্রাউজার Console ব্যবহার
▶

Chrome বা Firefox-এ F12 চাপুন → Console ট্যাবে যান। এখানেই JS কোড লিখে সাথে সাথে রান করা যায়।

console.log("Hello, World!");
// আউটপুট: Hello, World!
VS Code সেটআপ ও Live Server
▶

VS Code ডাউনলোড করুন। তারপর Live Server এক্সটেনশন ইনস্টল করুন — HTML ফাইল সেভ করলেই ব্রাউজারে অটো রিফ্রেশ হবে।

মোবাইলে Spck Editor অ্যাপ ব্যবহার করুন (Android/iOS)।

<!-- এক্সটার্নাল ফাইল -->

১ সপ্তাহ ▼
var, let, const গুরুত্বপূর্ণ
▶

let ও const ব্যবহার করুন — পুরনো var এড়িয়ে চলুন।

let naam = "Maidul";      // পরিবর্তনযোগ্য
const boyosh = 25;       // পরিবর্তন করা যাবে না
naam = "Rahim";           // ✓ ঠিক আছে
// boyosh = 30;         // ✗ Error হবে
ডেটা টাইপ: String, Number, Boolean, null, undefined
▶ 
let naam = "Maidul";   // String
let boyosh = 25;       // Number
let active = true;     // Boolean
let x = null;          // ইচ্ছাকৃতভাবে খালি
let y;                 // undefined — মান দেওয়া হয়নি
console.log(typeof naam); // "string"
গাণিতিক অপারেটর
▶ 
let a = 10, b = 3;
console.log(a + b);   // যোগ → 13
console.log(a - b);   // বিয়োগ → 7
console.log(a * b);   // গুণ → 30
console.log(a / b);   // ভাগ → 3.333
console.log(a % b);   // ভাগশেষ → 1
console.log(a ** b);  // ঘাত → 1000
তুলনামূলক অপারেটর (== vs ===) গুরুত্বপূর্ণ
▶

JS-এ === (strict) ব্যবহার করুন — টাইপ ও মান দুটোই মেলায়।

console.log(5 == "5");   // true (শুধু মান)
console.log(5 === "5");  // false (টাইপ আলাদা)
console.log(10 > 3);     // true
console.log(5 !== 3);    // true
লজিক্যাল অপারেটর (&&, ||, !)
▶ 
let x = 10;
console.log(x > 5 && x < 20); // উভয় সত্য → true
console.log(x > 5 || x > 100); // একটি সত্য → true
console.log(!(x > 5));          // উল্টো → false
Template Literal (ব্যাকটিক স্ট্রিং) গুরুত্বপূর্ণ
▶

Python-এর f-string-এর মতো — ব্যাকটিক ` ` দিয়ে ভেরিয়েবল সরাসরি টেক্সটে বসানো যায়।

let naam = "Maidul";
let boyosh = 25;
console.log(`আমার নাম ${naam}, বয়স ${boyosh}`);
// আউটপুট: আমার নাম Maidul, বয়স 25

কন্ট্রোল ফ্লো (Control Flow)শর্ত ও লুপ দিয়ে প্রোগ্রাম নিয়ন্ত্রণ

if / else if / else শর্ত গুরুত্বপূর্ণ
▶ 
let marks = 75;
if (marks >= 80) {
    console.log("A+");
} else if (marks >= 70) {
    console.log("A");   // এটি প্রিন্ট হবে
} else {
    console.log("B");
}
Ternary Operator (শর্টকাট শর্ত)
▶

এক লাইনে if-else লেখার উপায়।

let boyosh = 20;
let result = boyosh >= 18 ? "প্রাপ্তবয়স্ক" : "নাবালক";
console.log(result); // প্রাপ্তবয়স্ক
switch statement
▶ 
let day = "শুক্র";
switch (day) {
  case "শুক্র":
    console.log("ছুটির দিন!"); break;
  case "শনি":
    console.log("আধা ছুটি"); break;
  default:
    console.log("কর্মদিবস");
}
for লুপ গুরুত্বপূর্ণ
▶ 
for (let i = 1; i <= 5; i++) {
  console.log(`সংখ্যা: ${i}`);
}
// সংখ্যা: 1, সংখ্যা: 2 ... সংখ্যা: 5
while ও do...while লুপ
▶ 
let i = 1;
while (i <= 3) {
  console.log(i); // 1, 2, 3
  i++;
}

let j = 1;
do {
  console.log(j); // কমপক্ষে একবার রান হয়
  j++;
} while (j <= 3);
break ও continue
▶ 
for (let i = 1; i <= 5; i++) {
  if (i === 3) continue; // 3 বাদ দেয়
  if (i === 5) break;    // 5-এ থামে
  console.log(i); // 1, 2, 4
}

Arrow Function (=>)গুরুত্বপূর্ণ
▶

আধুনিক JS-এ সবচেয়ে বেশি ব্যবহৃত ফাংশন লেখার পদ্ধতি।

Default Parameter ও Rest Parameter
▶ 
function greet(naam = "বন্ধু") {
  console.log(`হ্যালো, ${naam}!`);
}
greet();            // হ্যালো, বন্ধু!
greet("Karim");    // হ্যালো, Karim!

function yog(...nums) {
  return nums.reduce((a, b) => a + b, 0);
}
console.log(yog(1,2,3,4)); // 10

Array তৈরি ও অ্যাক্সেস গুরুত্বপূর্ণ

Array Methods: map, filter, reduce গুরুত্বপূর্ণ
▶ 
let nums = [1, 2, 3, 4, 5];

// map — প্রতিটি উপাদান বদলায়
let double = nums.map(n => n * 2);
console.log(double); // [2,4,6,8,10]

// filter — শর্ত মেনে নেয়
let jor = nums.filter(n => n % 2 === 0);
console.log(jor);    // [2,4]

// reduce — একটি মানে পরিণত করে
let yog = nums.reduce((acc, n) => acc + n, 0);
console.log(yog);    // 15

Spread Operator (...)
▶

async / await গুরুত্বপূর্ণ
▶

Promise লেখার সহজ পদ্ধতি — অনেকটা সরল কোডের মতো দেখায়।

async function data_nao() {
  try {
    let res = await fetch("https://api.example.com/data");
    let data = await res.json();
    console.log(data);
  } catch (err) {
    console.log("ত্রুটি:", err);
  }
}
data_nao();
Fetch API দিয়ে ডেটা আনা গুরুত্বপূর্ণ
▶

ইন্টারনেট থেকে ডেটা নিয়ে আসার আধুনিক পদ্ধতি।

async function user_nao() {
  let res = await fetch("https://jsonplaceholder.typicode.com/users/1");
  let user = await res.json();
  console.log(user.name); // Leanne Graham
}
user_nao();

Inheritance (extends ও super)
▶ 
class Animal {
  constructor(naam) { this.naam = naam; }
  dakao() { console.log("..."); }
}

class Kukur extends Animal {
  dakao() { console.log("ঘেউ ঘেউ!"); }
}

let dog = new Kukur("Tommy");
dog.dakao(); // ঘেউ ঘেউ!
Prototype ও this কীওয়ার্ড অ্যাডভান্সড
▶

JS-এর প্রতিটি অবজেক্টের একটি Prototype থাকে — এটি JS-এর inheritance-এর মূল ভিত্তি।

function Student(naam) {
  this.naam = naam;
}
Student.prototype.greet = function() {
  console.log(`হ্যালো, আমি ${this.naam}`);
};

let s = new Student("Karim");
s.greet(); // হ্যালো, আমি Karim

Optional Chaining (?.) ও Nullish Coalescing (??)
▶ 
let user = { profile: { naam: "Rahim" } };

// ?. — না থাকলে error না দিয়ে undefined দেয়

// ?? — null/undefined হলে ডিফল্ট মান
let naam = user?.nickname ?? "অজানা";
console.log(naam); // অজানা

JS দিয়ে সার্ভার তৈরি করুন — একটাই ভাষায় ফ্রন্ট-ব্যাক দুটো!

const express = require("express");
const app = express();

app.get("/", (req, res) => {
  res.send("স্বাগতম আমার API-তে!");
});

app.listen(3000);
